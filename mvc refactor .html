Task Manager

    📦 Backend MVC Code (Reference)

    /* MODEL */
const taskSchema = new mongoose.Schema({ title: String, completed: Boolean });
const Task = mongoose.model("Task", taskSchema);

/* CONTROLLER */
const taskController = {
  getTasks: async (req, res) => { const tasks = await Task.find(); res.json(tasks); },
  createTask: async (req, res) => { const task = new Task(req.body); await task.save(); res.json(task); }
};

/* ROUTES */
app.get("/api/tasks", taskController.getTasks);
app.post("/api/tasks", taskController.createTask);
